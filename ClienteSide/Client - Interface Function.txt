-  ENG                         
----------------------------------------------------------------------------------------------------
OpenBrowser(link) -> open a link in browser

RenderImage(imageID, x, y, width, height) -> render image
RenderImage2(texture, x, y, w, h, u, v, uWidth, vHeight, scale, startScale, alpha) -> render bitmap with UV/scale/alpha
RenderImage3(texture, x, y, w, h, u, v, uWidth, vHeight, alpha) -> render image scale
RenderImageScale(texture, x, y, w, h, u, v, uWidth, vHeight, alpha) -> render image scale (same as RenderImage3)
RenderImageAuto(texture, x, y, w, h) -> render image auto scale

RenderText(x, y, text, width, sort) -> render text (color set with SetTextColor/SetTextBg)
RenderText2(x, y, text, width, sort) -> render text (wrapper)
RenderText3(x, y, text, sort) -> render text (wrapper)
RenderText4(x, y, text, sort) -> same as RenderText3
RenderText5(x, y, text, width, sort) -> same as RenderText2

SetFontType(type) -> set font type: 1 = Bold, 2 = Big, 0 = normal, 3 = Fix
SetTextBg(r, g, b, a) -> set text bg color (RGBA)
SetTextColor(r, g, b, a) -> set text color (RGBA)
GetTextPosY(font, text, y, height) -> return centered Y for text

CheckWindowOpen(windowID) -> return 1 if window is open
CloseWindow(windowID) -> close window
OpenWindow(windowID) -> open window

EnableAlphaTest() -> enable alpha test
DisableAlphaBlend() -> disable alpha blend
SetBlend() -> EnableAlphaTest(true)
GLSwitchBlend() -> enable alpha blend
GLSwitch() -> disable alpha blend

glColor3f(r, g, b) -> set OpenGL color
glColor4f(r, g, b, a) -> set OpenGL color RGBA

DrawBar(x, y, width, height) -> draw a colored rect/bar
EndDrawBar() -> end DrawBar

MousePosX() -> return mouse position X
MousePosY() -> return mouse position Y
ResetMouseL() -> reset left mouse flags
ResetMouseR() -> reset right mouse flags
ResetMouseM() -> reset middle mouse flags

CheckRepeatKey(key) -> return 1 if key is repeat
CheckPressedKey(key) -> return 1 if key is pressed
CheckReleasedKey(key) -> return 1 if key is released

CheckClickClient() -> return 1 if left click is active
DisableClickClient() -> disable clicks (interfaceLock=1) and reset mouse

WindowGetWidth() -> return window width (GetWindowsX)
WindowGetWidthAdd() -> return PositionX_In_The_Mid(0.0)
WindowGetHeight() -> return window height (GetWindowsY)
WindowGetHeighthAdd() -> return PositionY_In_The_Mid(0.0)
GetWideX() -> return PositionX_In_The_Mid(0.0)

GetImageWidth(texture) -> return real width from bitmap
GetImageHeight(texture) -> return real height from bitmap

ShowDescriptionComplete(x, y, itemIndex, level, dur, op1, op2, op3, exc, ancient, joh, socket, sock1, sock2, sock3, sock4, sock5) -> render item description complete
ShowItemDescription(x, y, itemIndex, level, op1, op2, op3, dur, exc, ancient) -> render item description

CreateItem(x, y, width, height, itemType, level, option1, extOption) -> render 3D item on screen/interface
DrawTooltip(x, y, text) -> draw simple tooltip text

SendMessageClient(text) -> send message in client chat

PlaySound(id) -> play sound
StopSound(id) -> stop sound

LockPlayerWalk() -> lock player walk
UnlockPlayerWalk() -> unlock player walk

SetLockInterfaces() -> lock interface click (ErrorMessage=1)
SetUnlockInterfaces() -> unlock interface click (ErrorMessage=0)

GetLanguage() -> return selected language string (Por/Eng/Spn)
